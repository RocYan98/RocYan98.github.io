int find_father(int x) {
    int a = x;
    while (x != father[x]) {
        x = father[x];
    }
    while (a != father[a]) {
        int tmp = a;
        a = father[a];
        father[tmp] = x;
    }
    return x;
}

void merge(int a, int b) {
    int fa = find_father(a);
    int fb = find_father(b);
    if (fa != fb) {
        father[fa] = fb;
    }
}
